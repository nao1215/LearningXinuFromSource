44 ## ソースコード
45 
46 ```
47  .text
48  .globl ctxsw
49 
50 ctxsw:
51  push {r0-r11, lr} /* r0〜r11およびlrをメモリに保存する */
52  push {lr} /* リターンアドレスを保存する */
53  mrs r2, cpsr /* cpsrコプロセッサからステータスを取得する */
54  push {r2} /* ステータスをメモリに保存する */
55  str sp, [r0] /* カレントプロセスのスタックポインタをメモリに保存する。*/
56  ldr sp, [r1] /* 新しいプロセスのスタックポインタをメモリから読み込む */
57  pop {r0} /* 引数としてステータスをメモリから取得する */
58  bl restore /* restore命令を呼び出し、リストアを行う */
59  pop {lr} /* リターンアドレスを取得する */
60  pop {r0-r12} /* 汎用レジスタをリストアする */
61  mov pc, r12 /* 新しいプロセスに戻る */
