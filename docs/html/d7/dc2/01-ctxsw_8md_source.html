16 ctxsw:

18  push {lr} /* リターンアドレスを保存する */
19  mrs r2, cpsr /* cpsrコプロセッサからステータスを取得する */

22  ldr sp, [r1] /* 新しいプロセスのスタックポインタをメモリから読み込む */
23  pop {r0} /* 引数としてステータスをメモリから取得する */
24  bl restore /* restore命令を呼び出し、リストアを行う */
25  pop {lr} /* リターンアドレスを取得する */
26  pop {r0-r12} /* 汎用レジスタをリストアする */
27  mov pc, r12 /* 新しいプロセスに戻る */
